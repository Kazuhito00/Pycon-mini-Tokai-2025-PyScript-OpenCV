import cv2

# モデル読み込み
net = cv2.dnn.readNetFromONNX("model.onnx")

# ビデオキャプチャ
cap = cv2.VideoCapture(0)

while True:
    ret, frame = cap.read()

    # 前処理
    blob = cv2.dnn.blobFromImage(frame, scalefactor=1/255.0, size=(224, 224), swapRB=True)
    net.setInput(blob)

    # 推論
    out = net.forward()
    # 後処理や表示など（中略）

cap.release()
cv2.destroyAllWindows()
